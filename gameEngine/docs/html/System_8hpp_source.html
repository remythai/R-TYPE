77/**
78 * @class System
79 * @brief CRTP (Curiously Recurring Template Pattern) base class for ECS
80 * systems.
81 *
82 * This template class provides a convenient base for defining game systems.
83 * Derived classes should implement:
84 * ```cpp
85 * void onUpdate(Registry& registry, float dt);
86 * ```
87 *
88 * Example usage:
89 * ```cpp
90 * class MovementSystem : public System<MovementSystem> {
91 * public:
92 * MovementSystem() {
93 * requireComponents<Transform, Velocity>();
94 * }
95 * void onUpdate(Registry& registry, float dt);
96 * };
97 * ```
98 *
99 * @tparam Derived The derived system type (used via CRTP).
100 */
101template <typename Derived>
102class System : public ISystem

104 public:

107 *

110 *
111 * @param registry ECS registry.
112 * @param dt Delta time since the last frame.
113 */

93 /**
94 * @brief Removes a component of type `Component` from entity `e`.
95 * @tparam Component Component type to remove.
96 * @param e Target entity.

98 template <typename Component>
99 void remove(Entity e)
100 {
101 ComponentID id =
102 ComponentRegistry::instance().getOrCreateID<Component>();
103
104 if (id < componentPools.size() && componentPools[id]) {
105 auto* pool = static_cast<ComponentPool<Component>*>(
106 componentPools[id].get());
107
108 if (pool->contains(e)) {
109 pool->erase(e);
110
111 if (pool->size() == 0) {
112 std::cout << "[Registry] Last component of type removed, "
113 "updating system availability\n";
114 availableComponents.reset(id);
115 updateSystemAvailability();
116 }
117 }

183 /**
184 * @brief Iterates through all entities that have all the specified
185 * components.
186 * @tparam Components Component types to include.
187 * @tparam Func Callable with signature `void(Entity, Components&...)`.
188 * @param func Function to execute for each matching entity.
189 */
190 template <typename... Components, typename Func>
191 void each(Func&& func)
192 {
193 IComponentPool* smallestPool = nullptr;
194 size_t smallestSize = SIZE_MAX;
195
196 (
197 (void)[&] {
198 ComponentID id =
199 ComponentRegistry::instance().getOrCreateID<Components>();
200 assurePool<Components>(id);
201
202 if (id < componentPools.size() && componentPools[id]) {
203 IComponentPool* pool = componentPools[id].get();
204 if (pool->size() < smallestSize) {
205 smallestSize = pool->size();
206 smallestPool = pool;
207 }
208 }
209 }(),
210 ...);
211
212 if (!smallestPool)
213 return;
214
215 for (size_t i = 0; i < smallestPool->size(); ++i) {
216 Entity e = smallestPool->getEntityAt(i);
217 if ((has<Components>(e) && ...)) {
218 func(e, get<Components>(e)...);
219 }
220 }
221 }

315 */
316 void updateSystemAvailability()
317 {
318 std::cout << "[Registry] Updating system availability...\n";
319 std::cout << " Available component types: "
320 << availableComponents.count() << "\n";
321
322 for (auto& system : systems) {
323 const auto& required = system->getSignature();
324 bool wasAvailable = system->hasRequiredComponents;
325
326 bool hasNoRequirements = required.none();
327 system->hasRequiredComponents =
328 hasNoRequirements ||
329 ((required & availableComponents) == required);
330
331 if (wasAvailable != system->hasRequiredComponents) {
332 std::cout << " System '" << system->getName() << "' is now "

470 /**
471 * @brief Ensures that a component pool exists for the given component ID.
472 * Creates it if necessary.
473 */
474 template <typename Component>
475 SparseSet<Entity, Component>& assurePool(ComponentID id)
476 {
477 if (id >= componentPools.size()) {
478 componentPools.resize(id + 1);
479 }
480
481 if (!componentPools[id]) {
482 componentPools[id] = std::make_unique<ComponentPool<Component>>();
483 }
484
485 return static_cast<ComponentPool<Component>*>(componentPools[id].get())
486 ->storage;
487 }

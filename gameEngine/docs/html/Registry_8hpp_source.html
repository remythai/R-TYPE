66 Component& emplace(Entity e, Args&&... args) {

70 bool wasEmpty = pool.size() == 0;

77 availableComponents.set(id);
78 updateSystemAvailability();

85 * @brief Removes a component of type `Component` from entity `e`.
86 * @tparam Component Component type to remove.
87 * @param e Target entity.

89 template<typename Component>
90 void remove(Entity e) {
91 ComponentID id = ComponentRegistry::instance().getOrCreateID<Component>();
92
93 if (id < componentPools.size() && componentPools[id]) {
94 auto* pool = static_cast<ComponentPool<Component>*>(componentPools[id].get());
95
96 if (pool->contains(e)) {
97 pool->erase(e);
98
99 // If no more of this component type exists, update system availability
100 if (pool->empty()) {
101 std::cout << "[Registry] Last component of type removed, updating system availability\n";
102 availableComponents.reset(id);
103 updateSystemAvailability();
104 }
105 }
106 }

158 /**
159 * @brief Iterates through all entities that have all the specified components.
160 * @tparam Components Component types to include.
161 * @tparam Func Callable with signature `void(Entity, Components&...)`.
162 * @param func Function to execute for each matching entity.
163 */
164 template<typename... Components, typename Func>
165 void each(Func&& func) {
166 IComponentPool* smallestPool = nullptr;
167 size_t smallestSize = SIZE_MAX;
168
169 // Find the smallest pool to optimize iteration
170 ((void)[&] {
171 ComponentID id = ComponentRegistry::instance().getOrCreateID<Components>();
172 assurePool<Components>(id);
173
174 if (id < componentPools.size() && componentPools[id]) {
175 IComponentPool* pool = componentPools[id].get();
176 if (pool->size() < smallestSize) {
177 smallestSize = pool->size();
178 smallestPool = pool;
179 }
180 }
181 }(), ...);
182
183 if (!smallestPool) return;
184
185 // Iterate and call function for entities that have all components
186 for (size_t i = 0; i < smallestPool->size(); ++i) {
187 Entity e = smallestPool->getEntityAt(i);
188 if ((has<Components>(e) && ...)) {
189 func(e, get<Components>(e)...);
190 }
191 }
192 }

194 /**

196 * @tparam SystemType The type of system to add.

198 * @param args Arguments forwarded to the system's constructor.

201 template<typename SystemType, typename... Args>

253 void update(float realDt) {

256 // Fixed time-step updates for deterministic simulation

272 /**
273 * @brief Updates which systems are active based on available component types.
274 */
275 void updateSystemAvailability() {
276 std::cout << "[Registry] Updating system availability...\n";
277 std::cout << " Available component types: " << availableComponents.count() << "\n";
278
279 for (auto& system : systems) {
280 const auto& required = system->getSignature();
281 bool wasAvailable = system->hasRequiredComponents;
282
283 // A system is active only if all its required components exist
284 system->hasRequiredComponents = (required & availableComponents) == required;
285
286 if (wasAvailable != system->hasRequiredComponents) {
287 std::cout << " System '" << system->getName()
288 << "' is now " << (system->hasRequiredComponents ? "ACTIVE" : "INACTIVE")
289 << "\n";
290 }
291 }
292 }

378 /**
379 * @brief Ensures that a component pool exists for the given component ID.
380 * Creates it if necessary.
381 */
382 template<typename Component>
383 SparseSet<Entity, Component>& assurePool(ComponentID id) {
384 if (id >= componentPools.size()) {
385 componentPools.resize(id + 1);
386 }
387
388 if (!componentPools[id]) {
389 componentPools[id] = std::make_unique<ComponentPool<Component>>();
390 }
391
392 return static_cast<ComponentPool<Component>*>(componentPools[id].get())->storage;
393 }

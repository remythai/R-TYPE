18 * @class MotionSystem
19 * @brief System responsible for detecting collisions and applying damage between entities.
20 *
21 * This system implements a simple **grid-based spatial partitioning** for collision detection.
22 * Entities are divided into grid cells based on their positions and collider sizes.
23 * Collisions are checked within each cell and neighboring cells.
24 *
25 * Entities must have the following components:
26 * - `Position` — entity position in 2D space.
27 * - `Renderable` — provides screen dimensions.
28 * - `Collider` — hitbox dimensions.
29 * - `Damage` — damage value dealt on collision.
30 * - `Health` — current hit points, reduced on collision.
31 */
32class MotionSystem : public System<MotionSystem> {
33private:
34 /**
35 * @brief Checks and resolves a collision between two entities.
36 *
37 * If their colliders overlap, applies damage to each entity accordingly,
38 * ensuring `Health.currentHp` does not drop below zero.
39 *
40 * @param e1 First entity.
41 * @param e2 Second entity.
42 * @param registry ECS registry for accessing components.
43 */
44 void collide(uint32_t e1, uint32_t e2, Registry& registry) {
45 Position e1Pos = registry.get<Position>(e1);
46 Position e2Pos = registry.get<Position>(e2);
47 Collider e1Hitbox = registry.get<Collider>(e1);
48 Collider e2Hitbox = registry.get<Collider>(e2);
49
50 Damage e1Damage = registry.get<Damage>(e1);
51 Damage e2Damage = registry.get<Damage>(e2);
52
53 Health e1Health = registry.get<Health>(e1);
54 Health e2Health = registry.get<Health>(e2);
55
56 // Axis-Aligned Bounding Box (AABB) collision check
57 if (e1Pos.x < e2Pos.x + e2Hitbox.size.x &&
58 e1Pos.x + e1Hitbox.size.x > e2Pos.x &&
59 e1Pos.y < e2Pos.y + e2Hitbox.size.y &&
60 e1Pos.y + e1Hitbox.size.y > e2Pos.y) {
61
62 // Apply damage safely, preventing negative HP
63 e1Health.currentHp = e1Health.currentHp > 0 ? std::max(e1Health.currentHp - e2Damage.dmg, 0) : e1Health.currentHp;
64 e2Health.currentHp = e2Health.currentHp > 0 ? std::max(e2Health.currentHp - e1Damage.dmg, 0) : e2Health.currentHp;
65 }
66 }

79 /**
80 * @brief Updates the system and processes collisions for all entities.
81 *
82 * Implements a grid-based spatial partitioning:
83 * - Divides the screen into grid cells of size `hitboxSizeMean`.
84 * - Each entity is assigned to cells based on its position and collider.
85 * - Collisions are checked within the cell and neighboring cells.
86 *
87 * @param registry The ECS registry managing entities and components.
88 * @param dt Delta time since the last update (unused here, but kept for consistency).
89 */
90 void onUpdate(Registry& registry, float dt) {
91 updateCount++;
92
93 Renderable screenInfo = registry.get<Renderable>(0);
94 uint32_t screenSizeX = screenInfo.screenSizeX;
95 uint32_t screenSizeY = screenInfo.screenSizeY;
96 uint32_t hitboxSizeMean = 64;
97
98 size_t gridWidth = screenSizeX / hitboxSizeMean;
99 size_t gridHeight = screenSizeY / hitboxSizeMean;
100
101 // 3D grid: grid[x][y] contains all entity IDs in that cell
102 std::vector<std::vector<std::vector<uint32_t>>> grid(
103 gridWidth, std::vector<std::vector<uint32_t>>(gridHeight));
104
105 // Populate grid with entities based on their positions and colliders
106 registry.each<Position, Renderable, Collider>(
107 [dt, &grid, hitboxSizeMean](auto e, Position& pos, Renderable& render, Collider& collider) {
108 float rightPos = pos.x + collider.size.x;
109 float topPos = pos.y + collider.size.y;
110
111 int leftCell = pos.x / hitboxSizeMean;
112 int bottomCell = pos.y / hitboxSizeMean;
113 int rightCell = rightPos / hitboxSizeMean;
114 int topCell = topPos / hitboxSizeMean;
115
116 for (size_t i = leftCell; i <= rightCell; i++) {
117 for (size_t j = bottomCell; j <= topCell; j++) {
118 grid[i][j].push_back(e);
119 }
120 }
121 }
122 );
123
124 // Check collisions within cells and neighboring cells
125 for (size_t i = 0; i < grid.size(); i++) {
126 for (size_t j = 0; j < grid[i].size(); j++) {
127 while (grid[i][j].size() > 1) {
128 for (size_t k = 1; k < grid[i][j].size(); k++) {
129 collide(grid[i][j][0], grid[i][j][k], registry);
130 }
131 // Neighboring cells
132 if (i + 1 < grid.size()) {
133 for (size_t k = 0; k < grid[i + 1][j].size(); k++) {
134 collide(grid[i][j][0], grid[i + 1][j][k], registry);
135 }
136 }
137 if (j + 1 < grid[i].size()) {
138 for (size_t k = 0; k < grid[i][j + 1].size(); k++) {
139 collide(grid[i][j][0], grid[i][j + 1][k], registry);
140 }
141 }
142 if (i + 1 < grid.size() && j + 1 < grid[i].size()) {
143 for (size_t k = 0; k < grid[i + 1][j + 1].size(); k++) {
144 collide(grid[i][j][0], grid[i + 1][j + 1][k], registry);
145 }
146 }
147 if (i > 0 && j + 1 < grid[i].size()) {
148 for (size_t k = 0; k < grid[i - 1][j + 1].size(); k++) {
149 collide(grid[i][j][0], grid[i - 1][j + 1][k], registry);
150 }
151 }
152 grid[i][j].erase(grid[i][j].begin());
153 }
154 }
155 }
156 }

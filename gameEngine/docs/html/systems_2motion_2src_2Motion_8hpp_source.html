11 *
12 * The `MotionSystem` updates all entities that have the following components:
13 * - `Position`
14 * - `Velocity`
15 * - `Acceleration`
16 * - `Renderable`
17 *
18 * It applies acceleration and deceleration to velocity, then integrates
19 * position accordingly while clamping movement within screen boundaries.
20 *
21 * ### Example Behavior:
22 * - Applies **deceleration** based on velocity and maximum speed.
23 * - Updates **velocity** according to acceleration and `dt`.
24 * - Updates **position** based on velocity and time.
25 * - Ensures position remains within the renderable area.
26 */

28 public:
29 /**

31 *
32 * Requires the following components for each processed entity:
33 * - `GameEngine::Position`
34 * - `GameEngine::Velocity`
35 * - `GameEngine::Acceleration`
36 * - `GameEngine::Renderable`

42 /**
43 * @brief Updates all entities matching the system’s component signature.
44 *
45 * For each entity:
46 * 1. Applies **deceleration** to gradually reduce velocity when acceleration is zero.
47 * 2. Applies **acceleration** to modify velocity.
48 * 3. Updates the **position** based on velocity and time delta.
49 * 4. Clamps the entity’s position within render bounds (`Renderable::screenSizeX/Y`).
50 *
51 * @param registry The ECS registry providing access to components.
52 * @param dt The elapsed time (in seconds) since the last update.
53 */
54 void onUpdate(Registry& registry, float dt) {
55 updateCount++;
56
57 registry.each<Position, Velocity, Acceleration, Renderable>(
58 [dt](auto e, Position& pos, Velocity& vel, Acceleration& acc, Renderable& render) {
59
60 // --- Deceleration ---
61 // Gradually slow down entity movement over time.
62 vel.x = vel.x > 0
63 ? std::max(vel.x - (vel.x / 5) - (vel.speedMax / 5), float(0))
64 : std::min(vel.x + (vel.x / 5) + (vel.speedMax / 5), float(0));
65
66 vel.y = vel.y > 0
67 ? std::max(vel.y - (vel.y / 5) - (vel.speedMax / 5), float(0))
68 : std::min(vel.y + (vel.y / 5) + (vel.speedMax / 5), float(0));
69
70 // --- Acceleration ---
71 // Update velocity within allowed speed range.
72 vel.x = std::clamp(vel.x + acc.x * dt, -vel.speedMax, vel.speedMax);
73 vel.y = std::clamp(vel.y + acc.y * dt, -vel.speedMax, vel.speedMax);
74
75 // --- Position Integration ---
76 // Apply velocity to position, respecting screen bounds.
77 pos.x = std::clamp(pos.x + vel.x * dt, float(0), render.screenSizeX);
78 pos.y = std::clamp(pos.y + vel.y * dt, float(0), render.screenSizeY);
79 }
80 );

11/**
12 * @class DomainHandler
13 * @brief System that manages entity domains and destroys entities outside their
14 * boundaries.
15 *
16 * The DomainHandler is an ECS system that monitors entities with Position and
17 * Domain components. It automatically destroys any entity that moves outside
18 * the boundaries defined by its Domain component.
19 *
20 * This system is useful for:
21 * - Removing projectiles that leave the game area
22 * - Cleaning up entities that go out of bounds
23 * - Enforcing level boundaries
24 *
25 * @details
26 * The system iterates through all entities that have both Position and Domain
27 * components. For each entity, it checks if the position (pos.x, pos.y) is
28 * within the domain boundaries:
29 * - **ax, ay**: Lower-left corner coordinates
30 * - **bx, by**: Upper-right corner coordinates
31 *
32 * If an entity is outside these boundaries, it is automatically destroyed.
33 *
34 * @note This system requires entities to have both Position and Domain
35 * components to be processed.
36 * @see Position
37 * @see Domain
38 */
39class DomainHandler : public System<DomainHandler>
40{
41 public:
42 /**
43 * @brief Constructs the DomainHandler system and sets up component
44 * requirements.
45 *
46 * This constructor registers the Position and Domain components as
47 * requirements for the system to be active and process entities.
48 *
49 * @post The system is configured to only process entities with Position and
50 * Domain components.
51 */
52 DomainHandler()
53 {
54 requireComponents<GameEngine::Position, GameEngine::Domain>();
55 }
56
57 /**
58 * @brief Updates the system, checking entity positions against their domain
59 * boundaries.
60 *
61 * This method is called each frame by the Registry. It:
62 * 1. Increments the internal update counter
63 * 2. Iterates through all entities with Position and Domain components
64 * 3. Checks if each entity's position is within its domain boundaries
65 * 4. Destroys any entity found outside its boundaries
66 *
67 * @param registry Reference to the main ECS registry for entity and
68 * component access.
69 * @param dt Delta time since the last update in seconds.
70 *
71 * @details
72 * An entity is considered outside its domain if:
73 * - `pos.pos.x < domain.ax` (left of lower-left corner)
74 * - `pos.pos.x > domain.bx` (right of upper-right corner)
75 * - `pos.pos.y < domain.ay` (below lower-left corner)
76 * - `pos.pos.y > domain.by` (above upper-right corner)
77 *
78 * @note Destroying entities during iteration is safe because the system
79 * uses the smallest pool for optimization, which doesn't invalidate
80 * iterators for subsequent entities.
81 *
82 * @see Registry::each
83 * @see Registry::destroy
84 */
85 void onUpdate(Registry& registry, float dt)
86 {
87 updateCount++;
88
89 registry.each<Position, Domain>(
90 [dt, &registry](auto e, Position& pos, Domain& domain) {
91 if (pos.pos.x < domain.ax || pos.pos.x > domain.bx ||
92 pos.pos.y < domain.ay || pos.pos.y > domain.by) {
93 registry.destroy(e);
94 }
95 });
96 }
97
98 /**
99 * @brief Counter tracking the number of update calls.
100 *
101 * This counter increments each time onUpdate() is called.
102 * Useful for debugging, profiling, or unit testing the system.
103 *
104 * @remarks This is a simple counter with no reset mechanism.
105 * Reset manually if needed for profiling purposes.
106 */
107 int updateCount = 0;
108};

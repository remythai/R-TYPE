7 * @struct GameClock
8 * @brief Fixed timestep simulation clock for consistent update timing.
9 *
10 * The `GameClock` manages time progression in a game or simulation loop.
11 * It maintains a fixed timestep (e.g., 1/60 seconds) for deterministic updates,
12 * while tracking total elapsed time, frame count, and interpolation alpha for rendering.
13 *
14 * This allows the game logic to run at a consistent rate independent of
15 * the actual frame rendering rate.
16 */
17 struct GameClock {
18 /// @brief Total accumulated simulation time (in seconds).
19 float totalTime = 0.0f;
20
21 /// @brief Fixed timestep duration (in seconds). Default = 1/60 = 0.01666...
22 float fixedDeltaTime = 1.0f / 60.0f;
23
24 /// @brief Number of fixed update frames that have occurred since start.
25 uint64_t frameCount = 0;
26
27 /// @brief Time scaling factor (1.0 = normal speed, 0.5 = half speed, 2.0 = double speed).
28 float timeScale = 1.0f;
29
30 /// @brief Accumulates real time between fixed updates.
31 float accumulator = 0.0f;
32
33 /**
34 * @brief Advances the internal clock based on real-world delta time.
35 *
36 * The function increments the accumulator using real time (`realDt * timeScale`)
37 * and determines how many fixed-timestep updates should occur during this frame.
38 *
39 * It caps the number of fixed updates per frame to prevent a "spiral of death"
40 * (where the simulation tries to catch up indefinitely after a large frame hitch).
41 *
42 * @param realDt Real-world delta time (seconds since last frame).
43 * @return The number of fixed simulation steps to perform this frame.
44 */
45 int update(float realDt) {
46 accumulator += realDt * timeScale;
47
48 int steps = 0;
49 const int maxSteps = 5; // Limit how many fixed steps can happen per frame
50
51 // Consume fixed steps from the accumulator
52 while (accumulator >= fixedDeltaTime && steps < maxSteps) {
53 totalTime += fixedDeltaTime;
54 frameCount++;
55 accumulator -= fixedDeltaTime;
56 steps++;
57 }
58
59 return steps;
60 }
61
62 /**
63 * @brief Returns the interpolation alpha between fixed updates.
64 *
65 * Used for smooth rendering between physics ticks.
66 * Commonly used in render interpolation as:
67 * `interpolated_position = lerp(previous, current, getInterpolationAlpha())`
68 *
69 * @return A value between 0.0 and 1.0 representing interpolation progress.
70 */
71 float getInterpolationAlpha() const {
72 return accumulator / fixedDeltaTime;
73 }
74
75 /**
76 * @brief Returns the fixed timestep scaled by the time scale.
77 *
78 * @return Scaled fixed delta time (seconds).
79 */
80 float getFixedDeltaTime() const {
81 return fixedDeltaTime * timeScale;
82 }
83 };

20/**
21 * @class InputHandler
22 * @brief System that processes player input and manages input-controlled
23 * entities.
24 *
25 * The InputHandler system translates input commands into entity behavior by:
26 * 1. Reading input states from InputControlled components
27 * 2. Updating entity acceleration based on directional inputs
28 * 3. Creating projectiles when shoot input is detected
29 *
30 * This system is responsible for converting raw input into game mechanics,
31 * serving as the bridge between player input and entity physics/behavior.
32 *
33 * @details
34 * **Input Mapping:**
35 * - Input 0: Move down (positive Y acceleration)
36 * - Input 1: Move up (negative Y acceleration)
37 * - Input 2: Move left (negative X acceleration)
38 * - Input 3: Move right (positive X acceleration)
39 * - Input 4: Shoot (creates projectile entity)
40 *
41 * **Projectile Creation:**
42 * When input 4 (shoot) is detected, the system:
43 * 1. Creates a new entity
44 * 2. Attaches required components (Renderable, Health, Damage, etc.)
45 * 3. Positions it at the player's current position
46 * 4. Sets up physics (Velocity, Acceleration)
47 * 5. Configures collision (Collider with bitset)
48 * 6. Restricts movement within domain boundaries
49 *
50 * @note Multiple inputs can be processed in a single frame, allowing
51 * simultaneous movement and shooting.
52 *
53 * @requires
54 * - InputControlled: Provides input commands
55 * - Acceleration: Modified by input
56 * - Renderable: Required for entity visibility (used for requirement checking)
57 * - Position: Used when creating projectiles
58 *
59 * @see InputControlled
60 * @see Acceleration
61 * @see Position
62 * @see Renderable
63 * @see Collider
64 * @see Domain
65 */

89 /**
90 * @brief Processes input commands and updates entity behavior accordingly.
91 *
92 * For each entity with InputControlled and Acceleration components:
93 * 1. Resets acceleration to zero
94 * 2. Iterates through pending input commands
95 * 3. Updates acceleration or spawns projectiles based on input type
96 * 4. Clears input queue after processing
97 *
98 * @param registry Reference to the ECS Registry for entity management.
99 * @param dt Delta time since last update in seconds (unused but required by
100 * interface).
101 *
102 * @details
103 * **Input Processing:**
104 * - Direction inputs (0-3) modify the Acceleration component
105 * - Shoot input (4) creates a new projectile entity with full setup
106 *
107 * **Projectile Properties:**
108 * - Position: Copied from player entity
109 * - Velocity: (10.0, 10.0) pixels per second
110 * - Acceleration: 10.0 (2D scalar)
111 * - Health: 1 HP (destroyed on first collision)
112 * - Damage: 1 damage per hit
113 * - Renderable: 22.28x22.28 sprite from
114 * "assets/sprites/playerProjectiles.png"
115 * - Collider: Bitset "01000000" (collision layer configuration)
116 * - Domain: Restricted to (0, 0) to (1905, 1080) screen boundaries
117 *
118 * **Acceleration Values:**
119 * - Directional movement: 5.0 units per second²
120 * - Movement range: ±5.0 on X and Y axes
121 *
122 * @attention
123 * - Acceleration is reset each frame, so movement inputs must be
124 * provided continuously to maintain motion.
125 * - Multiple conflicting direction inputs in the same frame will
126 * overwrite each other (last input wins).
127 *
128 * @note Projectile spawn position is retrieved fresh each frame,
129 * allowing for dynamic player movement while shooting.
130 *
131 * @example
132 * ```cpp
133 * // Player presses UP and SHOOT simultaneously
134 * // Result: acceleration.y = -5.0, projectile spawned at player pos
135 * ```
136 *
137 * @see Registry::each
138 * @see Registry::create
139 * @see Registry::emplace
140 * @see InputControlled
141 * @see Acceleration
142 */
143 void onUpdate(Registry& registry, float dt)
144 {
145 updateCount++;
146 registry.each<InputControlled, Acceleration>(
147 [dt, &registry](
148 auto e, InputControlled& inputs, Acceleration& acceleration) {
149 float accelerationValue = 5.0;
150 GameEngine::Position playerPos;
151 uint32_t shoot = -1;
152 acceleration.x = 0;
153 acceleration.y = 0;
154 std::vector<vec2> rectPos;
155
156 for (auto& it : inputs.inputs) {
157 switch (it) {
158 case 0:
159 /// @brief Move down
160 acceleration.y = accelerationValue;
161 break;
162 case 1:
163 /// @brief Move up
164 acceleration.y = -accelerationValue;
165 break;
166 case 2:
167 /// @brief Move left
168 acceleration.x = -accelerationValue;
169 break;
170 case 3:
171 /// @brief Move right
172 acceleration.x = accelerationValue;
173 break;
174 case 4:
175 /// @brief Shoot: Create projectile with full
176 /// component setup
177 shoot = registry.create();
178 rectPos.clear();
179 rectPos.push_back(vec2{0.0F, 0.0F});
180 rectPos.push_back(vec2{19.0F, 0.0F});
181 rectPos.push_back(vec2{38.0F, 0.0F});
182 registry.emplace<GameEngine::Renderable>(
183 shoot, 1920.0, 1080.0,
184 "assets/sprites/playerProjectiles.png", rectPos,
185 vec2{22.28f, 22.28f}, 50, true);
186 registry.emplace<GameEngine::Health>(shoot, 1, 1);
187 registry.emplace<GameEngine::Damage>(shoot, 1);
188 registry.emplace<GameEngine::Velocity>(
189 shoot, 10.0, 10.0);
190 registry.emplace<GameEngine::Acceleration>(
191 shoot, 10.0);
192 playerPos = registry.get<GameEngine::Position>(e);
193 registry.emplace<GameEngine::Position>(
194 shoot, playerPos.pos.x, playerPos.pos.y);
195 registry.emplace<GameEngine::Collider>(
196 shoot, vec2(0.0, 0.0),
197 std::bitset<8>("01000000"),
198 std::bitset<8>("00100000"), vec2(44.56, 44.56));
199 registry.emplace<GameEngine::Domain>(
200 shoot, 0, 0, 1905.0, 1080.0);
201 break;
202 default:
203 break;
204 }
205 }
206 });
207 }

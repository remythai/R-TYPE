200/**
201 * @brief Destroys a player entity and broadcasts the death event
202 *
203 * Removes the player entity from the ECS registry, clears the player slot,
204 * and sends a KILLED packet to all connected clients.
205 *
206 * @param playerId The ID of the player to destroy (0-3)
207 */
208void rtype::NetworkServer::destroyPlayerEntity(uint8_t playerId)
209{
210 std::lock_guard<std::mutex> lock(_registryMutex);
211 if (playerId < 4 &&
212 _playerSlots[playerId].entity != EntityManager::INVALID_ENTITY) {
213 EntityManager::Entity entityId = _playerSlots[playerId].entity;
214 std::string username = _playerSlots[playerId].username;
215
216 _registry->destroy(_playerSlots[playerId].entity);
217 _playerSlots[playerId].entity = EntityManager::INVALID_ENTITY;
218
219 std::vector<uint8_t> message;
220 message.push_back(static_cast<uint8_t>(rtype::PacketType::KILLED));
221
222 auto idBytes = toBytes<uint16_t>(0);
223 message.insert(message.end(), idBytes.begin(), idBytes.end());
224
225 auto now = std::chrono::steady_clock::now();
226 uint32_t timestamp =
227 std::chrono::duration_cast<std::chrono::milliseconds>(
228 now.time_since_epoch())
229 .count();
230 auto tsBytes = toBytes<uint32_t>(timestamp);
231 message.insert(message.end(), tsBytes.begin(), tsBytes.end());
232
233 message.push_back(static_cast<uint8_t>(entityId));
234
235 message.push_back(playerId);
236
237 uint8_t usernameLen = static_cast<uint8_t>(username.size());
238 message.push_back(usernameLen);
239
240 message.insert(message.end(), username.begin(), username.end());
241
242 broadcast(message);
243 }
244}

246/**
247 * @brief Applies input actions to a player's entity
248 *
249 * Updates the InputControlled component of the player's entity based on
250 * the received key code and action (press/release).
251 *
252 * @param playerId The ID of the player (0-3)
253 * @param keyCode The key code of the input action
254 * @param action The action type (1 for press, 0 for release)
255 */
256void rtype::NetworkServer::applyInputToEntity(
257 uint8_t playerId, uint8_t keyCode, uint8_t action)
258{
259 if (playerId >= 4 || !_playerSlots[playerId].isUsed)
260 return;
261
262 std::lock_guard<std::mutex> lock(_registryMutex);
263
264 auto entity = _playerSlots[playerId].entity;
265 if (entity == EntityManager::INVALID_ENTITY)
266 return;
267
268 if (!_registry->has<GameEngine::InputControlled>(entity))
269 return;
270
271 auto& inputCtrl = _registry->get<GameEngine::InputControlled>(entity);
272
273 if (action == 1) {
274 if (std::find(
275 inputCtrl.inputs.begin(), inputCtrl.inputs.end(), keyCode) ==
276 inputCtrl.inputs.end()) {
277 inputCtrl.inputs.push_back(keyCode);
278 }
279 } else if (action == 0) {
280 inputCtrl.inputs.erase(
281 std::remove(
282 inputCtrl.inputs.begin(), inputCtrl.inputs.end(), keyCode),
283 inputCtrl.inputs.end());
284 }
285}

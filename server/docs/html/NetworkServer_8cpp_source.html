16/**
17 * @brief Converts a float to a vector of 4 bytes
18 *
19 * @param value The float value to convert
20 * @return std::vector<uint8_t> A vector containing the 4 bytes representing the
21 * float
22 */
23std::vector<uint8_t> floatToBytes(float value)
24{
25 std::vector<uint8_t> bytes(4);
26 uint32_t temp;
27 std::memcpy(&temp, &value, sizeof(float));
28 bytes[0] = (temp >> 24) & 0xFF;
29 bytes[1] = (temp >> 16) & 0xFF;
30 bytes[2] = (temp >> 8) & 0xFF;
31 bytes[3] = temp & 0xFF;
32 return bytes;
33}

124/**
125 * @brief Runs the network server
126 *
127 * Starts the server, handles incoming packets, updates the ECS,
128 * checks for inactive players, and broadcasts snapshots.
129 */
130void rtype::NetworkServer::run()
131{
132 _running = true;
133 _lastSnapshot = std::chrono::steady_clock::now();
134
135 if (_game == "RType" && !_mapPath.empty()) {
136 if (loadEnemiesFromJson(_mapPath) == 84) {
137 _running = false;
138 return;
139 }
140 }
141
142 doReceive();
143 std::cout << "UDP Server running..." << std::endl;
144
145 std::thread([this]() {
146 while (_running) {
147 cleanInactivePlayers();
148 std::this_thread::sleep_for(std::chrono::seconds(1));
149 }
150 }).detach();
151
152 std::thread([this]() {
153 const float targetDt = 1.0f / 120.0f;
154 auto lastUpdateTime = std::chrono::steady_clock::now();
155 auto clock1 = std::chrono::steady_clock::now();
156 auto clock2 = std::chrono::steady_clock::now();
157
158 while (_running) {
159 auto frameStart = std::chrono::steady_clock::now();
160
161 float realDt =
162 std::chrono::duration<float>(frameStart - lastUpdateTime)
163 .count();
164 lastUpdateTime = frameStart;
165
166 realDt = std::min(realDt, 0.25f);
167
168 _gameTime += realDt;
169
170 if (_game == "RType")
171 checkAndSpawnEnemies();
172
173 updateECS(realDt);
174
175 auto frameEnd = std::chrono::steady_clock::now();
176 float elapsed =
177 std::chrono::duration<float>(frameEnd - frameStart).count();
178
179 if (elapsed < targetDt) {
180 std::this_thread::sleep_for(
181 std::chrono::duration<float>(targetDt - elapsed));
182 }
183
184 clock2 = std::chrono::steady_clock::now();
185 auto deltaTime = clock2 - clock1;
186 if (std::chrono::duration_cast<std::chrono::milliseconds>(deltaTime)
187 .count() > 5000) {
188 this->createEnemyEntity();
189 clock1 = std::chrono::steady_clock::now();
190 }
191 }
192 }).detach();
193
194 std::thread([this]() {
195 while (_running) {
196 auto now = std::chrono::steady_clock::now();
197 float elapsed =
198 std::chrono::duration<float>(now - _lastSnapshot).count();
199
200 if (elapsed >= SNAPSHOT_RATE) {
201 broadcastSnapshot();
202 _lastSnapshot = now;
203 }
204
205 std::this_thread::sleep_for(std::chrono::milliseconds(10));
206 }
207 }).detach();
208
209 _ioContext.run();
210}

212/**
213 * @brief Cleans up inactive players who have timed out
214 *
215 * Checks each player slot for inactivity exceeding 30 seconds.
216 * Destroys the player's entity, clears the slot, and broadcasts a TIMEOUT
217 * packet.
218 */
219void rtype::NetworkServer::cleanInactivePlayers()
220{
221 std::lock_guard<std::mutex> lock(_playerSlotsMutex);
222
223 auto now = std::chrono::steady_clock::now();
224 for (auto& slot : _playerSlots) {
225 if (slot.isUsed) {
226 auto duration = std::chrono::duration_cast<std::chrono::seconds>(
227 now - slot.lastActive)
228 .count();
229 if (duration > 30) {
230 uint8_t playerId = slot.playerId;
231 EntityManager::Entity entityId = slot.entity;
232 std::string username = slot.username;
233
234 {
235 std::lock_guard<std::mutex> regLock(_registryMutex);
236 if (entityId != EntityManager::INVALID_ENTITY) {
237 _registry->destroy(entityId);
238 slot.entity = EntityManager::INVALID_ENTITY;
239 }
240 }
241
242 slot.isUsed = false;
243
244 std::vector<uint8_t> message;
245 message.push_back(
246 static_cast<uint8_t>(rtype::PacketType::TIMEOUT));
247
248 auto idBytes = toBytes<uint16_t>(0);
249 message.insert(message.end(), idBytes.begin(), idBytes.end());
250
251 uint32_t timestamp =
252 std::chrono::duration_cast<std::chrono::milliseconds>(
253 now.time_since_epoch())
254 .count();
255 auto tsBytes = toBytes<uint32_t>(timestamp);
256 message.insert(message.end(), tsBytes.begin(), tsBytes.end());
257
258 message.push_back(static_cast<uint8_t>(entityId));
259 message.push_back(playerId);
260
261 uint8_t usernameLen = static_cast<uint8_t>(username.size());
262 message.push_back(usernameLen);
263 message.insert(message.end(), username.begin(), username.end());
264
265 broadcast(message);
266
267 std::cout << "[SERVER] Player " << int(playerId) << " ("
268 << username
269 << ") timed out. Entity: " << int(entityId)
270 << std::endl;
271 }
272 }
273 }

276/**
277 * @brief Asynchronously receives incoming UDP packets
278 *
279 * Sets up an asynchronous receive operation on the UDP socket.
280 * Upon receiving a packet, it extracts the packet type, ID, timestamp,
281 * and payload, then dispatches it to the appropriate handler.
282 */
283void rtype::NetworkServer::doReceive()
284{
285 auto buffer = std::make_shared<std::vector<uint8_t>>(1024);
286 auto clientEndpoint = std::make_shared<asio::ip::udp::endpoint>();
287
288 _socket.async_receive_from(
289 asio::buffer(*buffer), *clientEndpoint,
290 [this, buffer, clientEndpoint](
291 std::error_code ec, std::size_t bytesReceived) {
292 if (!ec && bytesReceived >= 7) {
293 PacketType type = static_cast<PacketType>((*buffer)[0]);
294 uint16_t packetId = fromBytes<uint16_t>(buffer->data() + 1);
295 uint32_t timestamp = fromBytes<uint32_t>(buffer->data() + 3);
296
297 size_t payloadSize = bytesReceived - 7;
298 std::vector<uint8_t> payload(payloadSize);
299 std::copy(
300 buffer->begin() + 7, buffer->begin() + 7 + payloadSize,
301 payload.begin());
302
303 handleClientPacket(
304 *clientEndpoint, type, packetId, timestamp, payload);
305 }
306 if (_running)
307 doReceive();
308 });
309}

393/**
394 * @brief Serializes the current ECS state into a snapshot packet
395 *
396 * Gathers the state of all entities with Renderable and Position components
397 * and constructs a snapshot packet to be sent to clients.
398 *
399 * @return std::vector<uint8_t> The serialized snapshot packet
400 */
401std::vector<uint8_t> rtype::NetworkServer::serializeSnapshot()
402{
403 std::lock_guard<std::mutex> lock(_registryMutex);
404 std::vector<uint8_t> snapshot;
405 snapshot.push_back(static_cast<uint8_t>(PacketType::SNAPSHOT));
406
407 auto idBytes = toBytes<uint16_t>(0);
408 snapshot.insert(snapshot.end(), idBytes.begin(), idBytes.end());
409
410 auto now = std::chrono::steady_clock::now();
411 uint32_t timestamp = std::chrono::duration_cast<std::chrono::milliseconds>(
412 now.time_since_epoch())
413 .count();
414 auto tsBytes = toBytes<uint32_t>(timestamp);
415 snapshot.insert(snapshot.end(), tsBytes.begin(), tsBytes.end());
416
417 _registry->each<GameEngine::Renderable, GameEngine::Position>(
418 [&](EntityManager::Entity entity, GameEngine::Renderable& render,
419 GameEngine::Position& pos) {
420 snapshot.push_back(static_cast<uint8_t>(entity));
421
422 auto xBytes = floatToBytes(pos.pos.x);
423 snapshot.insert(snapshot.end(), xBytes.begin(), xBytes.end());
424 auto yBytes = floatToBytes(pos.pos.y);
425 snapshot.insert(snapshot.end(), yBytes.begin(), yBytes.end());
426
427 uint8_t pathLen =
428 static_cast<uint8_t>(render.spriteSheetPath.size());
429 snapshot.push_back(pathLen);
430 snapshot.insert(
431 snapshot.end(), render.spriteSheetPath.begin(),
432 render.spriteSheetPath.end());
433
434 auto rectPosX = floatToBytes(render.currentRectPos.x);
435 snapshot.insert(snapshot.end(), rectPosX.begin(), rectPosX.end());
436 auto rectPosY = floatToBytes(render.currentRectPos.y);
437 snapshot.insert(snapshot.end(), rectPosY.begin(), rectPosY.end());
438
439 auto rectSizeX = floatToBytes(render.rectSize.x);
440 snapshot.insert(snapshot.end(), rectSizeX.begin(), rectSizeX.end());
441 auto rectSizeY = floatToBytes(render.rectSize.y);
442 snapshot.insert(snapshot.end(), rectSizeY.begin(), rectSizeY.end());
443 });
444
445 return snapshot;
446}

14/**
15 * @brief Trims whitespace from the beginning and end of a string
16 *
17 * @param str The string to trim
18 * @return std::string The trimmed string
19 */
20static std::string trim(const std::string& str)
21{
22 size_t first = str.find_first_not_of(" \t\n\r");
23 if (first == std::string::npos)
24 return "";
25 size_t last = str.find_last_not_of(" \t\n\r");
26 return str.substr(first, last - first + 1);
27}

29/**
30 * @brief Extracts a JSON value for a given key within a specified range
31 *
32 * @param content The JSON content as a string
33 * @param key The key to search for
34 * @param start The starting position in the content
35 * @param end The ending position in the content
36 * @return std::string The extracted and trimmed value
37 */
38static std::string extractJSONValue(
39 const std::string& content, const std::string& key, size_t start,
40 size_t end)
41{
42 size_t keyPos = content.find("\"" + key + "\"", start);
43 if (keyPos >= end)
44 return "";
45
46 size_t colonPos = content.find(':', keyPos);
47 if (colonPos >= end)
48 return "";
49
50 size_t commaPos = content.find(',', colonPos);
51 size_t bracePos = content.find('}', colonPos);
52 size_t valueEnd =
53 (commaPos < bracePos && commaPos < end) ? commaPos : bracePos;
54
55 return trim(content.substr(colonPos + 1, valueEnd - colonPos - 1));
56}

58/**
59 * @brief Parses a JSON string value for a given key
60 *
61 * @param content The JSON content as a string
62 * @param key The key to search for
63 * @param start The starting position in the content
64 * @param end The ending position in the content
65 * @return std::string The extracted string value without quotes
66 */
67static std::string parseJSONString(
68 const std::string& content, const std::string& key, size_t start,
69 size_t end)
70{
71 size_t keyPos = content.find("\"" + key + "\"", start);
72 if (keyPos >= end)
73 return "";
74
75 size_t colonPos = content.find(':', keyPos);
76 size_t quoteStart = content.find('"', colonPos + 1);
77 size_t quoteEnd = content.find('"', quoteStart + 1);
78
79 return (quoteStart < end && quoteEnd < end)
80 ? content.substr(quoteStart + 1, quoteEnd - quoteStart - 1)
81 : "";
82}

84/**
85 * @brief Parses a JSON array of floats for a given key
86 *
87 * @param content The JSON content as a string
88 * @param key The key to search for
89 * @param start The starting position in the content
90 * @param end The ending position in the content
91 * @return std::array<float, 4> Array containing the parsed float values (x, y,
92 * width, height)
93 */
94static std::array<float, 4> parseJSONArray(
95 const std::string& content, const std::string& key, size_t start,
96 size_t end)
97{
98 std::array<float, 4> result = {0.0f, 0.0f, 33.0f, 36.0f};
99
100 size_t keyPos = content.find("\"" + key + "\"", start);
101 if (keyPos >= end)
102 return result;
103
104 size_t bracketStart = content.find('[', keyPos);
105 size_t bracketEnd = content.find(']', bracketStart);
106 if (bracketStart >= end || bracketEnd >= end)
107 return result;
108
109 std::string arrayStr =
110 content.substr(bracketStart + 1, bracketEnd - bracketStart - 1);
111 std::stringstream ss(arrayStr);
112 std::string item;
113 int index = 0;
114
115 while (std::getline(ss, item, ',') && index < 4) {
116 result[index++] = std::stof(trim(item));
117 }
118
119 return result;
120}

122/**
123 * @brief Loads enemy spawn data from a JSON file
124 *
125 * Parses a JSON file containing enemy entities and populates the spawn list.
126 * Enemies are sorted by spawn time after loading.
127 *
128 * @param filepath Path to the JSON file containing enemy data
129 */
130int rtype::NetworkServer::loadEnemiesFromJson(const std::string& filepath)
131{
132 std::ifstream file(filepath);
133 if (!file.is_open()) {
134 throw NetworkServerError("Could not open enemy file : " + filepath);
135 return 84;
136 }
137
138 std::string content(
139 (std::istreambuf_iterator<char>(file)),
140 std::istreambuf_iterator<char>());
141 file.close();
142
143 _enemySpawnList.clear();
144
145 size_t entitiesPos = content.find("\"entities\"");
146 if (entitiesPos == std::string::npos) {
147 throw NetworkServerError("No 'entities' array found in JSON");
148 return 84;
149 }
150
151 size_t arrayStart = content.find('[', entitiesPos);
152 size_t arrayEnd = content.rfind(']');
153 if (arrayStart == std::string::npos || arrayEnd == std::string::npos) {
154 throw NetworkServerError("Invalid JSON format");
155 return 84;
156 }
157
158 size_t pos = arrayStart + 1;
159 while (pos < arrayEnd) {
160 size_t objStart = content.find('{', pos);
161 if (objStart >= arrayEnd)
162 break;
163
164 size_t objEnd = content.find('}', objStart);
165 if (objEnd >= arrayEnd)
166 break;
167
168 EnemySpawnData enemy;
169 enemy.type =
170 std::stoi(extractJSONValue(content, "type", objStart, objEnd));
171 enemy.x = std::stof(extractJSONValue(content, "x", objStart, objEnd));
172 enemy.y = std::stof(extractJSONValue(content, "y", objStart, objEnd));
173 enemy.spawnTime =
174 std::stof(extractJSONValue(content, "spawnTime", objStart, objEnd));
175 enemy.spritePath =
176 parseJSONString(content, "spritePath", objStart, objEnd);
177 enemy.textureRect =
178 parseJSONArray(content, "textureRect", objStart, objEnd);
179
180 _enemySpawnList.push_back(enemy);
181
182 pos = objEnd + 1;
183 }
184
185 std::sort(
186 _enemySpawnList.begin(), _enemySpawnList.end(),
187 [](const EnemySpawnData& a, const EnemySpawnData& b) {
188 return a.spawnTime < b.spawnTime;
189 });
190
191 std::cout << "[SERVER] Loaded " << _enemySpawnList.size()
192 << " enemies from " << filepath << std::endl;
193 return 0;
194}

198 *
199 * Iterates through the spawn list and creates enemies whose spawn time

202void rtype::NetworkServer::checkAndSpawnEnemies()
203{
204 while (_nextEnemyToSpawn < _enemySpawnList.size()) {
205 const EnemySpawnData& enemyData = _enemySpawnList[_nextEnemyToSpawn];
206

209 _nextEnemyToSpawn++;
